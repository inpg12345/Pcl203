#include <stdio.h>
#define ROW1 3
#define COL1 2
#define ROW2 2
#define COL2 3

int main()
{
    int x, y, z;
    int macierz[ROW1][COL2]={{0,0,0},{0,0,0},{0,0,0}};
    int macierz1[ROW1][COL1]={{1,2},{2,3},{3,4}};
    int macierz2[ROW2][COL2]={{1,2,3},{3,2,1}};
    
    for(x=0; x<ROW1; x++)
    {
        for(y=0; y<COL2; y++)
        {
            for(z=0; z<COL1; z++)
            {
                macierz[x][y]+=macierz1[x][z]*macierz2[z][y];
            }    
        }    
    }
    
    for(x=0; x<ROW1; x++)
    {
        for(y=0; y<COL2; y++)
        {
            printf("%d ", macierz[x][y]);
        }
        putchar('\n');
    }
    return 0;
}
